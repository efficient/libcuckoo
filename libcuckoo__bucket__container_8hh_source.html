165  void swap(libcuckoo_bucket_container &bc) noexcept {
166  swap_allocator(allocator_, bc.allocator_,
167  typename traits_::propagate_on_container_swap());
168  bucket_allocator_ = allocator_;
169  // Regardless of whether we actually swapped the allocators or not, it will
170  // always be okay to do the remainder of the swap. This is because if the
171  // allocators were swapped, then the subsequent operations are okay. If the
172  // allocators weren't swapped but compare equal, then we're okay. If they
173  // weren't swapped and compare unequal, then behavior is undefined, so
174  // we're okay.
175  size_t bc_hashpower = bc.hashpower();
176  bc.hashpower(hashpower());
177  hashpower(bc_hashpower);
178  std::swap(buckets_, bc.buckets_);
179  }

196  // Constructs live data in a bucket
197  template <typename K, typename... Args>
198  void setKV(size_type ind, size_type slot, partial_t p, K &&k,
199  Args &&... args) {
200  bucket &b = buckets_[ind];
201  assert(!b.occupied(slot));
202  b.partial(slot) = p;
203  traits_::construct(allocator_, std::addressof(b.storage_kvpair(slot)),
204  std::piecewise_construct,
205  std::forward_as_tuple(std::forward<K>(k)),
206  std::forward_as_tuple(std::forward<Args>(args)...));
207  // This must occur last, to enforce a strong exception guarantee
208  b.occupied(slot) = true;
209  }

211  // Destroys live data in a bucket
212  void eraseKV(size_type ind, size_type slot) {
213  bucket &b = buckets_[ind];
214  assert(b.occupied(slot));
215  b.occupied(slot) = false;
216  traits_::destroy(allocator_, std::addressof(b.storage_kvpair(slot)));
217  }

219  // Destroys all the live data in the buckets. Does not deallocate the bucket
220  // memory.
221  void clear() noexcept {
222  static_assert(
223  std::is_nothrow_destructible<key_type>::value &&
224  std::is_nothrow_destructible<mapped_type>::value,
225  "libcuckoo_bucket_container requires key and value to be nothrow "
226  "destructible");
227  for (size_type i = 0; i < size(); ++i) {
228  bucket &b = buckets_[i];
229  for (size_type j = 0; j < SLOT_PER_BUCKET; ++j) {
230  if (b.occupied(j)) {
231  eraseKV(i, j);
232  }
233  }
234  }
235  }

280  void destroy_buckets() noexcept {
281  if (buckets_ == nullptr) {
282  return;
283  }
284  // The bucket default constructor is nothrow, so we don't have to
285  // worry about dealing with exceptions when constructing all the
286  // elements.
287  static_assert(std::is_nothrow_destructible<bucket>::value,
288  "libcuckoo_bucket_container requires bucket to be nothrow "
289  "destructible");
290  clear();
291  for (size_type i = 0; i < size(); ++i) {
292  traits_::destroy(allocator_, &buckets_[i]);
293  }
294  bucket_allocator_.deallocate(buckets_, size());
295  buckets_ = nullptr;
296  }

311  template <bool B>
312  bucket *transfer(
313  size_type dst_hp,
314  typename std::conditional<B, libcuckoo_bucket_container &,
315  const libcuckoo_bucket_container &>::type src,
316  std::integral_constant<bool, B> move) {
317  assert(dst_hp >= src.hashpower());
318  libcuckoo_bucket_container dst(dst_hp, get_allocator());
319  // Move/copy all occupied slots of the source buckets
320  for (size_t i = 0; i < src.size(); ++i) {
321  for (size_t j = 0; j < SLOT_PER_BUCKET; ++j) {
322  if (src.buckets_[i].occupied(j)) {
323  dst.move_or_copy(i, j, src.buckets_[i], j, move);
324  }
325  }
326  }
327  // Take away the pointer from `dst` and return it
328  bucket *dst_pointer = dst.buckets_;
329  dst.buckets_ = nullptr;
330  return dst_pointer;
331  }

347  // If the key and value are Trivial, the bucket be serilizable. Since we
348  // already disallow user-specialized instances of std::pair, we know that the
349  // default implementation of std::pair uses a default copy constructor, so
350  // this should be okay. We could in theory just check if the type is
351  // TriviallyCopyable but this check is not available on some compilers we
352  // want to support.
353  template <typename Bogus = void *>
354  friend typename std::enable_if<sizeof(Bogus) && std::is_trivial<Key>::value &&
355  std::is_trivial<T>::value,
356  std::ostream &>::type
357  operator<<(std::ostream &os, const libcuckoo_bucket_container &bc) {
358  size_type hp = bc.hashpower();
359  os.write(reinterpret_cast<const char *>(&hp), sizeof(size_type));
360  os.write(reinterpret_cast<const char *>(bc.buckets_),
361  sizeof(bucket) * bc.size());
362  return os;
363  }
