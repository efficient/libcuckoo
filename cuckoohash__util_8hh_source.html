90 // Allocates an array of the given size and value-initializes each element with

92 template <class T, class Alloc>
93 T* create_array(const size_t size) {
94  Alloc allocator;
95  T* arr = allocator.allocate(size);
96  // Initialize all the elements, safely deallocating and destroying
97  // everything in case of error.
98  size_t i;
99  try {
100  for (i = 0; i < size; ++i) {
101  allocator.construct(&arr[i]);
102  }
103  } catch (...) {
104  for (size_t j = 0; j < i; ++j) {
105  allocator.destroy(&arr[j]);
106  }
107  allocator.deallocate(arr, size);
108  throw;
109  }
110  return arr;
111 }

124 // executes the function over the given range split over num_threads threads
125 template <class F>
126 static void parallel_exec(size_t start, size_t end,
127  size_t num_threads, F func) {
128  size_t work_per_thread = (end - start) / num_threads;
129  std::vector<std::thread> threads(num_threads);
130  for (size_t i = 0; i < num_threads - 1; ++i) {
131  threads[i] = std::thread(func, start, start + work_per_thread);
132  start += work_per_thread;
133  }
134  threads[num_threads - 1] = std::thread(func, start, end);
135  for (std::thread& t : threads) {
136  t.join();
137  }
138 }
